// ══════════════════════════════════════════════════════════════ // FESTIVAL CAD DISPATCH // ────────────────────────────────────────────────────────────── // First time: run setupTriggers once, then run buildCADSheet // Every time after: just run buildCADSheet // Never edit this script. // All your data lives in the Config sheet. // ══════════════════════════════════════════════════════════════ function buildCADSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── Config sheet ─────────────────────────────────────────────── let config = ss.getSheetByName("Config"); if (!config) { config = ss.insertSheet("Config"); setupConfig(config); } // ── Dispatch sheet ───────────────────────────────────────────── let dispatch = ss.getSheetByName("Dispatch"); if (!dispatch) { dispatch = ss.insertSheet("Dispatch"); } else { dispatch.clear(); dispatch.clearConditionalFormatRules(); dispatch.clearNotes(); } // ── Meta sheet (hidden — stores row numbers for fast triggers) ─ let meta = ss.getSheetByName("_meta"); if (!meta) { meta = ss.insertSheet("_meta"); meta.hideSheet(); } else { meta.clear(); } // ── Read Config ──────────────────────────────────────────────── const allConfig = config.getDataRange().getValues(); const posts = getConfigList(allConfig, 0); const units = getConfigList(allConfig, 1); const statuses = getConfigList(allConfig, 2); // ── Calculate row numbers ────────────────────────────────────── const S1_DATA = 3; const S1_LAST = S1_DATA + posts.length - 1; const S2_HEAD = S1_LAST + 3; const S2_DATA = S2_HEAD + 2; const S2_LAST = S2_DATA + units.length - 1; // ── Store row numbers in meta sheet ─────────────────────────── meta.getRange(1, 1, 4, 2).setValues([ ["S1_DATA", S1_DATA], ["S1_LAST", S1_LAST], ["S2_DATA", S2_DATA], ["S2_LAST", S2_LAST], ]); // ── Section 1 title and headers ──────────────────────────────── dispatch.getRange(1, 1) .setValue("SECTION 1 — POST LOCATIONS") .setFontWeight("bold"); dispatch.getRange(2, 1, 1, 4) .setValues([["Post Location", "Unit / Callsign", "Status", "Timestamp"]]) .setFontWeight("bold"); // ── Section 1 dropdowns and pre-filled posts ─────────────────── dispatch.getRange(S1_DATA, 1, posts.length, 1) .setDataValidation( SpreadsheetApp.newDataValidation() .requireValueInRange(config.getRange("A2:A" + (posts.length + 1)), true) .setAllowInvalid(false).build() ) .setValues(posts.map(p => [p])); dispatch.getRange(S1_DATA, 2, posts.length, 1) .setDataValidation( SpreadsheetApp.newDataValidation() .requireValueInRange(config.getRange("B2:B" + (units.length + 1)), true) .setAllowInvalid(false).build() ); dispatch.getRange(S1_DATA, 3, posts.length, 1) .setDataValidation( SpreadsheetApp.newDataValidation() .requireValueInRange(config.getRange("C2:C" + (statuses.length + 1)), true) .setAllowInvalid(false).build() ); // ── Section 2 title and headers ──────────────────────────────── dispatch.getRange(S2_HEAD, 1) .setValue("SECTION 2 — UNIT ROSTER") .setFontWeight("bold"); dispatch.getRange(S2_HEAD + 1, 1, 1, 4) .setValues([["Unit / Callsign", "Current Post", "Status", "Timestamp"]]) .setFontWeight("bold"); // ── Section 2 unit dropdown and pre-filled units ─────────────── dispatch.getRange(S2_DATA, 1, units.length, 1) .setDataValidation( SpreadsheetApp.newDataValidation() .requireValueInRange(config.getRange("B2:B" + (units.length + 1)), true) .setAllowInvalid(false).build() ) .setValues(units.map(u => [u])); // ── Section 2 formulas ───────────────────────────────────────── units.forEach((unit, i) => { const r = S2_DATA + i; dispatch.getRange(r, 2).setFormula( `=IFERROR(INDEX(A${S1_DATA}:A${S1_LAST},MATCH(A${r},B${S1_DATA}:B${S1_LAST},0)),"Unassigned")` ); dispatch.getRange(r, 3).setFormula( `=IFERROR(INDEX(C${S1_DATA}:C${S1_LAST},MATCH(A${r},B${S1_DATA}:B${S1_LAST},0)),"—")` ); }); // ── Freeze header row ────────────────────────────────────────── dispatch.setFrozenRows(2); ss.setActiveSheet(dispatch); SpreadsheetApp.getUi().alert( "Done!\n\n" + posts.length + " post locations\n" + units.length + " units\n" + statuses.length + " status codes\n\n" + "To update: edit the Config sheet then run buildCADSheet again." ); } // ══════════════════════════════════════════════════════════════ // RUN ONCE — sets up the edit and change triggers // Only need to run this again if triggers stop working // ══════════════════════════════════════════════════════════════ function setupTriggers() { const ss = SpreadsheetApp.getActiveSpreadsheet(); ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t)); ScriptApp.newTrigger("onEditHandler").forSpreadsheet(ss).onEdit().create(); ScriptApp.newTrigger("onChangeHandler").forSpreadsheet(ss).onChange().create(); SpreadsheetApp.getUi().alert("Triggers set up successfully."); } // ══════════════════════════════════════════════════════════════ // TRIGGERS // ══════════════════════════════════════════════════════════════ // Fires on every direct cell edit function onEditHandler(e) { const ss = e.source; const sheet = e.range.getSheet(); const row = e.range.getRow(); const col = e.range.getColumn(); if (sheet.getName() !== "Dispatch") return; const m = getMeta(ss); if (!m) return; const now = Utilities.formatDate( new Date(), ss.getSpreadsheetTimeZone(), "HH:mm:ss" ); // S1 col C (status) changed — stamp or clear S1 col D if (col === 3 && row >= m.S1_DATA && row <= m.S1_LAST) { const status = sheet.getRange(row, 3).getValue(); sheet.getRange(row, 4).setValue(status !== "" ? now : ""); return; } // S1 col B (unit) cleared — clear S1 col D if (col === 2 && row >= m.S1_DATA && row <= m.S1_LAST) { const unit = sheet.getRange(row, 2).getValue(); if (unit === "") sheet.getRange(row, 4).clearContent(); return; } // S2 col A (unit) added or changed — stamp S2 col D if (col === 1 && row >= m.S2_DATA && row <= m.S2_LAST) { const unit = sheet.getRange(row, 1).getValue(); sheet.getRange(row, 4).setValue(unit !== "" ? now : ""); return; } } // Fires when any formula recalculates — catches S2 col C updating function onChangeHandler(e) { const ss = e.source; const sheet = ss.getSheetByName("Dispatch"); if (!sheet) return; const m = getMeta(ss); if (!m) return; const now = Utilities.formatDate( new Date(), ss.getSpreadsheetTimeZone(), "HH:mm:ss" ); // Read all of S2 in one batch call const s2Rows = sheet .getRange(m.S2_DATA, 1, m.S2_LAST - m.S2_DATA + 1, 3) .getValues(); // Stamp col D for any unit that has a real status s2Rows.forEach((row, i) => { const unit = row[0]; const status = row[2]; if (unit !== "" && status !== "—" && status !== "") { sheet.getRange(m.S2_DATA + i, 4).setValue(now); } }); } // ══════════════════════════════════════════════════════════════ // HELPERS // ══════════════════════════════════════════════════════════════ // Reads a column from Config, skipping the header row function getConfigList(data, colIndex) { const results = []; for (let i = 1; i < data.length; i++) { const val = String(data[i][colIndex]).trim(); if (val !== "" && val !== "undefined") results.push(val); } return results; } // Reads the four stored row numbers from the hidden _meta sheet function getMeta(ss) { const meta = ss.getSheetByName("_meta"); if (!meta) return null; const vals = meta.getRange(1, 1, 4, 2).getValues(); return { S1_DATA: vals[0][1], S1_LAST: vals[1][1], S2_DATA: vals[2][1], S2_LAST: vals[3][1], }; } // Builds the Config sheet the first time only function setupConfig(sheet) { sheet.getRange(1, 1).setValue("POSTS").setFontWeight("bold"); sheet.getRange(1, 2).setValue("UNITS").setFontWeight("bold"); sheet.getRange(1, 3).setValue("STATUSES").setFontWeight("bold"); const posts = [ "Main Stage", "North Gate", "South Gate", "Medical Tent", "VIP Area", "Parking Lot A", "Artist Village", "West Perimeter", "Command Post", ]; const units = [ "U-01", "U-02", "U-03", "U-04", "U-05", "U-06", "U-07", ]; const statuses = [ "10-8", "10-6", "10-7", "10-97", "10-4", ]; sheet.getRange(2, 1, posts.length, 1).setValues(posts.map(p => [p])); sheet.getRange(2, 2, units.length, 1).setValues(units.map(u => [u])); sheet.getRange(2, 3, statuses.length, 1).setValues(statuses.map(s => [s])); }
